interceptarAlert.js

Uma biblioteca JavaScript para interceptar e customizar alertas, prompts e confirms padrão do navegador.

Instalação

Clique na caixa abaixo para copiar o código:

<script>
const GITHUB_URL = "https://raw.githubusercontent.com/cmstudioapps/Bibliotecas/main/interceptarAlert.js";
const LIB_KEY = "interceptarAlertLib";

function loadCustomAlert() {
const storedLib = localStorage.getItem(LIB_KEY);

if (storedLib) {
eval(storedLib);
} else {
fetch(GITHUB_URL)
.then(response => response.text())
.then(script => {
localStorage.setItem(LIB_KEY, script);
eval(script);
})
.catch(error => alert("Erro ao carregar a biblioteca: " + error.message));
}
}
loadCustomAlert();
</script>

Uso

Clique na caixa abaixo para copiar o código:

alert("Isso é um alerta customizado!");

Uso do Prompt

Clique na caixa abaixo para copiar o código:

const resposta = await prompt("Qual é o seu nome?", "Nome padrão");
console.log(resposta);

Uso do Confirm

Clique na caixa abaixo para copiar o código:

const resposta = await confirm("Você tem certeza?");
console.log(resposta);
